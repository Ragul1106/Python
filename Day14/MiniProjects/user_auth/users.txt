import hashlib
import os
from datetime import datetime

USERS_FILE = "users.txt"
LOG_FILE = "login_log.txt"

def hash_password(password):
    return hashlib.sha256(password.encode()).hexdigest()

def signup():
    username = input("Choose a username: ").strip()
    password = input("Choose a password: ").strip()
    hashed = hash_password(password)

    if os.path.exists(USERS_FILE):
        with open(USERS_FILE, "r") as f:
            for line in f:
                if line.startswith(username + ":"):
                    print("‚ùå Username already exists.")
                    return

    with open(USERS_FILE, "a") as f:
        f.write(f"{username}:{hashed}\n")
    print("‚úÖ Signup successful!")

def login():
    username = input("Username: ").strip()
    password = input("Password: ").strip()
    hashed_input = hash_password(password)

    if not os.path.exists(USERS_FILE):
        print("‚ùå No users found. Please signup first.")
        return

    success = False
    with open(USERS_FILE, "r") as f:
        for line in f:
            saved_username, saved_hash = line.strip().split(":")
            if username == saved_username and hashed_input == saved_hash:
                success = True
                break

    log_attempt(username, success)

    if success:
        print("‚úÖ Login successful!")
    else:
        print("‚ùå Invalid credentials.")

def log_attempt(username, success):
    with open(LOG_FILE, "a") as log:
        status = "SUCCESS" if success else "FAILED"
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        log.write(f"[{timestamp}] Login attempt by '{username}': {status}\n")

def main():
    while True:
        print("\n==== User Auth System ====")
        print("1. Signup")
        print("2. Login")
        print("3. Exit")

        choice = input("Enter choice: ")
        if choice == "1":
            signup()
        elif choice == "2":
            login()
        elif choice == "3":
            print("üëã Exiting.")
            break
        else:
            print("‚ùå Invalid choice.")

if __name__ == "__main__":
    main()
